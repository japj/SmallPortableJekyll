def self.descendants
  descendants = []
  ObjectSpace.each_object(singleton_class) do |k|
    descendants.unshift k unless k == self
  end
  descendants
end

def self.priority(priority = nil)
  @priority ||= nil
  if priority && PRIORITIES.key?(priority)
    @priority = priority
  end
  @priority || :normal
end
