def set_form(easy)
  easy.url ||= url
  form.params_encoding = params_encoding
  if form.multipart?
    form.escape = false
    form.materialize
    easy.httppost = form.first.read_pointer
  else
    form.escape = easy.escape?
    easy.postfieldsize = form.to_s.bytesize
    easy.copypostfields = form.to_s
  end
end
