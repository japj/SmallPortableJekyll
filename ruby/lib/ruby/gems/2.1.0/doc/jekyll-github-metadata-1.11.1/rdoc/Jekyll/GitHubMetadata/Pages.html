  else

  end

def env_var(key, intermediate_default = nil)
  !ENV[key].to_s.empty? ? ENV[key] : (intermediate_default || DEFAULTS[key])
end

    # File lib/jekyll-github-metadata/pages.rb, line 72
def trim_last_slash(url)
  if url[-1] == "/"
    url[0..-2]
  else
    url
  end
end
