def create_commit_comment(repo, sha, body, path=nil, line=nil, position=nil, options = {})
  params = {
    :body => body,
    :path => path,
    :line => line,
    :position => position
  }
  post "#{Repository.path repo}/commits/#{sha}/comments", options.merge(params)
end

def update_commit_comment(repo, id, body, options = {})
