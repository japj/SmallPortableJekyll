def list(content, list_type)
  case list_type
  when :ordered
    "\n\n.nr step 0 1\n#{content}\n"
  when :unordered
    "\n.\n#{content}\n"
  end
end

def list_item(content, list_type)
  case list_type
  when :ordered
    ".IP \\n+[step]\n#{content.strip}\n"
  when :unordered
    ".IP \\[bu] 2 \n#{content.strip}\n"
  end
end
