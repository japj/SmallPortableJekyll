def initialize(name_with_owner, access_token: nil)
  unless name_with_owner =~ REPO_REGEX
    raise Errors::InvalidRepositoryError
  end
  parts = name_with_owner.split("/")
  @owner = parts.first
  @name  = parts.last
  @access_token = access_token || ENV["OCTOKIT_ACCESS_TOKEN"]
end
