def render(context)
  @context = context
  attrs    = attributes.map { |k, v| %Q(#{k}="#{v}") }.join(' ')
  "<link #{attrs} />"
end

    :href => "#{url}/#{path}",
    :title => title
  }.keep_if { |_, v| v }

def url
  if config["url"]
    URI.join(config["url"], config["baseurl"])
  elsif config["github"] && config["github"]["url"]
    config["github"]["url"]
  end
end
